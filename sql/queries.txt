//nested selection, projection -- traceemployee.php
SELECT employee.ename, employee.ephone
FROM employee
WHERE employee.eid like (SELECT deliveryorder.eid
                        FROM deliveryorder
                        WHERE order_id like '%$order_id%') ;

//join -- employeeworkarea.php

FROM employee
INNER JOIN employeeworkarea ON employee.eid=employeeworkarea.eid
ORDER BY employee.ename ASC ;
